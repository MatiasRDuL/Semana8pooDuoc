    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton1ActionPerformed
        
        try{
            String nombre = txtnombre.getText().trim();
            String direccion = txtdireccion.getText().trim();
            String datosAdicionales = txtdatos.getText().trim();

            if (nombre.isEmpty() || direccion.isEmpty() ||  datosAdicionales.isEmpty()) {
                javax.swing.JOptionPane.showMessageDialog(this, "Ingrese todos los campos", "Error", JOptionPane.ERROR_MESSAGE);
                return;
            }

            String tipoEntidad = (String) jComboBox1.getSelectedItem();

            if ("Centro de cultivo".equals(tipoEntidad)) {
                int toneladasProduccion = Integer.parseInt(datosAdicionales);
                CentroCultivo centro = new CentroCultivo(nombre, direccion, toneladasProduccion);
                gestor.agregarRegistrable(centro);
                cargarDatos();
                javax.swing.JOptionPane.showMessageDialog(this, "Centro de cultivo registrado exitosamente", "Éxito", JOptionPane.INFORMATION_MESSAGE);
            } else if ("Planta de proceso".equals(tipoEntidad)) {
                int capacidadProceso = Integer.parseInt(datosAdicionales);
                PlantaProceso planta =  new PlantaProceso(nombre, direccion, capacidadProceso);
                gestor.agregarRegistrable(planta);
                cargarDatos();
                javax.swing.JOptionPane.showMessageDialog(this, "Planta de proceso registrada exitosamente", "Éxito", JOptionPane.INFORMATION_MESSAGE);
            }  else if ("Empleado".equals(tipoEntidad)) {
                Empleado empleado = new Empleado(nombre, direccion, datosAdicionales);
                gestor.agregarRegistrable(empleado);
                cargarDatos();
                javax.swing.JOptionPane.showMessageDialog(this, "Empleado registrado exitosamente", "Éxito", JOptionPane.INFORMATION_MESSAGE);
            }
            
        }catch(Exception e){
            javax.swing.JOptionPane.showMessageDialog(this, "Error al agregar Centro de cultivo" + e.getMessage(), "Error", JOptionPane.ERROR_MESSAGE);
        }
// TODO add your handling code here:
    }//GEN-LAST:event_jButton1ActionPerformed

    private void actualizarCampossegunEntidad() {
        String tipoEntindad = (String) jComboBox1.getSelectedItem();
        txtdatos.setText("");
        if ("Centro de cultivo".equals(tipoEntindad)) {
            jLabel3.setText("Toneladas de Produccion");
        } else if ("Planta de proceso".equals(tipoEntindad)) {
            jLabel3.setText("Capacidad de proceso diario en toneladas");
        } else if ("Empleado".equals(tipoEntindad)) {
            jLabel3.setText("Cargo del empleado");
        }
    }

    private void jComboBox1ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jComboBox1ActionPerformed
        actualizarCampossegunEntidad();
    }//GEN-LAST:event_jComboBox1ActionPerformed

    private void tablaConfiguracion() {
        DefaultTableModel modelo = new DefaultTableModel(new Object[][] {},new String[] {"Nombre","Direccion","Datos adicionales"}){
            @Override
            public boolean isCellEditable(int rowIndex, int columnIndex) {
                return false;
            }
        };
        jTable1.setModel(modelo);
    }

    private void cargarDatos() {
        DefaultTableModel modelo = (DefaultTableModel) jTable1.getModel();
        modelo.setRowCount(0);

        for (Registrable entidad: gestor.obtenerTodas()){
            Entidad e = (Entidad) entidad;

            Object[] fila = new Object[3];
            fila[0] = e.getNombre();
            fila[1] = e.getDireccion();
            fila[2] = entidad.getResumen();

            modelo.addRow(fila);
        }
    }
